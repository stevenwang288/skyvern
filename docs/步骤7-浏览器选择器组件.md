```typescript
      // 尝试使用自定义协议打开
      window.open('adspower://', '_blank');
    } catch {
      // 降级到官网下载页面
      window.open('https://www.adspower.com/download', '_blank');
    }
  };

  // 自动填充Chrome路径
  const autoFillChromePath = () => {
    if (browserSuggestions?.recommended_paths?.length > 0) {
      setValue(`${name}.chrome_path`, browserSuggestions.recommended_paths[0]);
    }
  };

  // 添加常用Chrome参数
  const addCommonChromeArg = (arg: string) => {
    const currentArgs = watchedConfig.chrome_args || [];
    if (!currentArgs.includes(arg)) {
      setValue(`${name}.chrome_args`, [...currentArgs, arg]);
    }
  };
```

```typescript
      // 验证配置格式
      if (config.type && Object.values(BrowserType).includes(config.type)) {
        Object.keys(config).forEach(key => {
          setValue(`${name}.${key}`, config[key]);
        });
      } else {
        alert('无效的配置文件格式');
      }
    } catch (error) {
      alert('配置文件解析失败');
    }
  };
  reader.readAsText(file);
};
```
